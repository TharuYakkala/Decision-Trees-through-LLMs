def func(row):
    node1 = int(row['Official'] == 1)
    node2 = int(row['Round'] <= 6)
    node3 = int(row['Judge'] == 'USA Today')
    emb = [node1, node2, node3]
    if node1 == 1:
        if node2 == 1:
            return 0, emb
        else:
            return 1, emb
    else:
        if node3 == 1:
            return 0, emb
        else:
            return 1, emb
