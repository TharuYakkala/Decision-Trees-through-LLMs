def func(row):
    node1 = int(row['Judge'] <= 5)
    node2 = int(row['Official'] == 1)
    node3 = int(row['Round'] <= 6)
    emb = [node1, node2, node3]
    if node1 == 1:
        if node2 == 1:
            return 0, emb
        else:
            if node3 == 1:
                return 0, emb
            else:
                return 1, emb
    else:
        if node2 == 1:
            return 1, emb
        else:
            if node3 == 1:
                return 1, emb
            else:
                return 0, emb
