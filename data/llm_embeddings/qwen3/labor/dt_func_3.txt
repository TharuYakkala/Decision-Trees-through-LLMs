    node1 = int(row['cost of living adjustment'] != 'none')
    node2 = int(row['vacation'] == 'gnr')
    node3 = int(row['pension'] != 'none')
    node4 = int(row['wage increase first year'] > 5)
    node5 = int(row['standby pay'] > 10)
    node6 = int(row['education allowance'] == 'true')
    node7 = int(row['statutory holidays'] > 12)
    emb = [node1, node2, node3, node4, node5, node6, node7]
    
    if node1 == 0:
        return 0, emb
    if node2 == 1:
        return 1, emb
    if node3 == 0:
        return 0, emb
    if node4 == 1:
        return 1, emb
    if node5 == 1:
        return 1, emb
    if node6 == 1:
        return 1, emb
    if node7 == 1:
        return 1, emb
    return 0, emb
