def func(row):
    # Precompute binary vector embeddings
    node1 = int(row['oldpeak'] <= 2.0)
    node2 = int(row['thal'] <= 2)
    node3 = int(row['sex'] == 'male')
    emb = [node1, node2, node3]
    
    # Decision tree logic
    if node1 == 1:
        if node2 == 1:
            if node3 == 1:
                prediction = 0
            else:
                prediction = 1
        else:
            prediction = 1
    else:
        prediction = 1
    
    return prediction, emb
