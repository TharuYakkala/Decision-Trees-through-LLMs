def func(row):
    node1 = int(row['chest_pain'] == 0)
    node2 = int(row['oldpeak'] > 2.5)
    node3 = int(row['thalach'] < 120)
    node4 = int(row['sex'] == 'male')
    node5 = int(row['trestbps'] > 140)
    node6 = int(row['chol'] > 240)
    emb = [node1, node2, node3, node4, node5, node6]
    if node1 == 1:
        prediction = 0
    else:
        if node2 == 1:
            prediction = 1
        else:
            if node3 == 1:
                prediction = 1
            else:
                if node5 == 1:
                    prediction = 1
                else:
                    if node6 == 1:
                        prediction = 1
                    else:
                        prediction = node4
    return prediction, emb
