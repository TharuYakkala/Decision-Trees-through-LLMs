def penguin_classifier(row):
    node1 = int(row['flipper_length_mm'] <= 200)
    node2 = int(row['culmen_length_mm'] <= 38)
    node3 = int(row['body_mass_g'] > 3700)
    emb = [node1, node2, node3]
    
    if node1 == 1:
        if node2 == 1:
            prediction = 0  # Adelie
        else:
            prediction = 1  # Chinstrap
    else:
        if node3 == 1:
            prediction = 2  # Gentoo
        else:
            prediction = 2  # Gentoo
    
    return prediction, emb
