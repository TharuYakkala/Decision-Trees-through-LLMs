def func(row):
    node1 = int(row['flipper_length_mm'] > 200)
    node2 = int(row['culmen_length_mm'] < 38)
    node3 = int(row['culmen_depth_mm'] < 17)
    emb = [node1, node2, node3]
    if node1:
        prediction = 2  # Gentoo
    else:
        if node2:
            prediction = 0  # Adelie
        else:
            if node3:
                prediction = 1  # Chinstrap
            else:
                prediction = 0  # Adelie
    return prediction, emb
