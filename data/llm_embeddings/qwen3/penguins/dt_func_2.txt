def func(row):
    node1 = int(row['body_mass_g'] > 4000)
    node2 = int(row['flipper_length_mm'] < 190)
    node3 = int(row['culmen_depth_mm'] < 16)
    emb = [node1, node2, node3]
    if emb[0] == 1:
        return 2, emb
    elif emb[1] == 1:
        return 0, emb
    elif emb[2] == 1:
        return 1, emb
    else:
        return 0, emb
