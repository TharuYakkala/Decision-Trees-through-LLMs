def func(row):
    node1 = int(row['BVE/BVL'] <= 1.0)
    node2 = int(row['WC/TA'] <= 10.0)
    node3 = int(row['EBIT/TA'] <= 5.0)
    node4 = int(row['S/TA'] <= 20.0)
    node5 = int(row['RE/TA'] <= 5.0)
    emb = [node1, node2, node3, node4, node5]
    if node1 == 1:
        if node2 == 1:
            if node3 == 1:
                if node4 == 1:
                    if node5 == 1:
                        return 1, emb
                    else:
                        return 0, emb
                else:
                    return 0, emb
            else:
                return 0, emb
        else:
            return 0, emb
    else:
        return 0, emb
