def func(row):
    node1 = int(row['WC/TA'] <= 0.10)
    node2 = int(row['RE/TA'] <= 0.05)
    node3 = int(row['EBIT/TA'] <= 0.05)
    node4 = int(row['S/TA'] <= 0.20)
    node5 = int(row['BVE/BVL'] <= 0.50)
    emb = [node1, node2, node3, node4, node5]
    
    if node1 == 1:
        if node2 == 1:
            prediction = 1
        else:
            prediction = 0
    else:
        if node3 == 1:
            if node4 == 1:
                prediction = 1
            else:
                prediction = 0
        else:
            if node5 == 1:
                prediction = 1
            else:
                prediction = 0
    return prediction, emb
