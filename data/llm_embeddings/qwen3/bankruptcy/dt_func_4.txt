def func(row):
    node1 = int(row['BVE/BVL'] <= 0.5)
    node2 = int(row['EBIT/TA'] <= 5)
    node3 = int(row['WC/TA'] <= 5)
    node4 = int(row['S/TA'] <= 10)
    emb = [node1, node2, node3, node4]
    
    if node1 == 1:
        return 1, emb
    else:
        if node2 == 1:
            if node3 == 1:
                return 1, emb
            else:
                return 0, emb
        else:
            if node4 == 1:
                return 1, emb
            else:
                return 0, emb
