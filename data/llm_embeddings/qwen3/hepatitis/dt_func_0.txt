def func(row):
    node1 = int(row['BILIRUBIN'] > 2.0)
    node2 = int(row['ALBUMIN'] < 3.0)
    node3 = int(row['PROTIME'] > 15)
    node4 = int(row['ASCITES'] == 1)
    node5 = int(row['HISTOLOGY'] == 1)
    node6 = int(row['VARICES'] == 1)
    emb = [node1, node2, node3, node4, node5, node6]
    
    if node1 == 1:
        return 0, emb
    elif node2 == 1:
        if node3 == 1:
            return 0, emb
        elif node4 == 1:
            return 0, emb
        elif node5 == 1:
            return 0, emb
        else:
            return 1, emb
    elif node6 == 1:
        return 0, emb
    else:
        return 1, emb
