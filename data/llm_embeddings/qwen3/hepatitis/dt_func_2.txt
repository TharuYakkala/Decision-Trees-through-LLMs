def func(row):
    node1 = int(row['BILIRUBIN'] > 2.0)
    node2 = int(row['ALBUMIN'] < 3.0)
    node3 = int(row['PROTIME'] > 15)
    node4 = int(row['HISTOLOGY'] == 1)
    node5 = int(row['ASCITES'] == 1)
    emb = [node1, node2, node3, node4, node5]
    if node1 == 1:
        return 0, emb
    if node2 == 1:
        if node3 == 1:
            return 0, emb
        else:
            if node4 == 1:
                return 0, emb
            else:
                return 1, emb
    else:
        if node5 == 1:
            return 0, emb
        else:
            return 1, emb
