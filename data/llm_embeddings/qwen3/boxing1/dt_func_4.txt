def decision_tree(row):
    node1 = int(row['Official'] == 1)
    node2 = int(row['Judge'] == 8)  # ESPN
    node3 = int(row['Round'] <= 6)
    node4 = int(row['Round'] <= 4)
    emb = [node1, node2, node3, node4]
    
    if node1 == 1:
        if node2 == 1:
            prediction = 1
        else:
            if node3 == 1:
                prediction = 1
            else:
                prediction = 0
    else:
        if node4 == 1:
            prediction = 0
        else:
            prediction = 1
    return prediction, emb
