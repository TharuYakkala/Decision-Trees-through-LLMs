def func(row):
    node1 = int(row['Official'] == 1)
    node2 = int(row['Round'] <= 6)
    emb = [node1, node2]
    if node1 == 1:
        if node2 == 1:
            prediction = 1
        else:
            prediction = 0
    else:
        if node2 == 1:
            prediction = 0
        else:
            prediction = 1
    return prediction, emb
