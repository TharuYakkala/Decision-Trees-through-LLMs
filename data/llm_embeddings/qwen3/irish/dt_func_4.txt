def func(row):
    node1 = int(row['Educational_level'] <= 5)
    node2 = int(row['Prestige_score'] >= 15)
    node3 = int(row['DVRT'] >= 50)
    node4 = int(row['Type_school'] == 1)
    node5 = int(row['Sex'] == 1)
    emb = [node1, node2, node3, node4, node5]
    if node1 == 1:
        if node2 == 1:
            prediction = 1
        else:
            if node3 == 1:
                prediction = 1
            else:
                if node4 == 1:
                    prediction = 1
                else:
                    if node5 == 1:
                        prediction = 1
                    else:
                        prediction = 0
    else:
        prediction = 1
    return prediction, emb
