def func(row):
    node1 = int(row['Educational_level'] <= 5)
    node2 = int(row['DVRT'] > 20)
    node3 = int(row['Prestige_score'] > 15)
    node4 = int(row['Type_school'] == 0)
    node5 = int(row['Sex'] == 1)
    emb = [node1, node2, node3, node4, node5]
    
    if node1:
        if node2:
            prediction = 1
        else:
            if node3:
                prediction = 1
            else:
                prediction = 0
    else:
        if node4:
            prediction = 0
        else:
            if node5:
                prediction = 1
            else:
                prediction = 0
    
    return prediction, emb
