def func(row):
    node1 = int(row['Educational_level'] == 0)
    node2 = int(row['DVRT'] > 15)
    node3 = int(row['Prestige_score'] > 15)
    node4 = int(row['Type_school'] in [1, 2])
    emb = [node1, node2, node3, node4]
    if node1 == 1:
        return 1, emb
    if node2 == 1:
        if node3 == 1:
            return 1, emb
        else:
            if node4 == 1:
                return 1, emb
            else:
                return 0, emb
    else:
        return 0, emb
