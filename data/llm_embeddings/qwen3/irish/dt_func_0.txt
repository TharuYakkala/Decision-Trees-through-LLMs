def func(row):
    node1 = int(row['Educational_level'] <= 0)
    node2 = int(row['Type_school'] == 1)
    node3 = int(row['Prestige_score'] > 15)
    node4 = int(row['DVRT'] > 12)
    emb = [node1, node2, node3, node4]
    if node1 == 1:
        prediction = 1
    else:
        if node2 == 1:
            if node3 == 1:
                prediction = 1
            else:
                if node4 == 1:
                    prediction = 1
                else:
                    prediction = 0
        else:
            if node4 == 1:
                prediction = 1
            else:
                prediction = 0
    return prediction, emb
