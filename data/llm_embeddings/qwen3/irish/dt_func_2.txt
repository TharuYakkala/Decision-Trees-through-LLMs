def func(row):
    node1 = int(row['Educational_level'] <= 3)
    node2 = int(row['Type_school'] <= 0.5)
    node3 = int(row['Prestige_score'] > 15)
    node4 = int(row['DVRT'] > 10)
    node5 = int(row['Prestige_score'] > 20)
    emb = [node1, node2, node3, node4, node5]
    
    if node1 == 1:
        if node2 == 1:
            prediction = 0
        else:
            if node3 == 1:
                prediction = 1
            else:
                prediction = 0
    else:
        if node4 == 1:
            prediction = 1
        else:
            if node5 == 1:
                prediction = 1
            else:
                prediction = 0
    return prediction, emb
