def func(row):
    node1 = int(row['EBIT/TA'] <= 0.1)
    node2 = int(row['NI/TC'] <= 0.05)
    node3 = int(row['WC/TA'] <= 0.2)
    node4 = int(row['Equity/TL'] <= 0.5)
    emb = [node1, node2, node3, node4]
    if node1 == 0:
        return 1, emb
    if node2 == 0:
        return 1, emb
    if node3 == 0:
        return 1, emb
    if node4 == 0:
        return 1, emb
    return 0, emb
