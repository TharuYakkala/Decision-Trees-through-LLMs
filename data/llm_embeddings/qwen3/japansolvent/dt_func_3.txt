    node2 = int(row['WC/TA'] > 0.3)
    node3 = int(row['EBIT/TA'] > 0.1)
    node4 = int(row['NI/Sales'] > 0.05)
    emb = [node1, node2, node3, node4]

    elif node3 == 1:
        return 1, emb
