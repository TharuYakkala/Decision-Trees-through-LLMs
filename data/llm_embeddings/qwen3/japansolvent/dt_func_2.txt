def func(row):
    node1 = int(row['WC/TA'] <= 0.3)
    node2 = int(row['Equity/TL'] <= 0.5)
    node3 = int(row['NI/TC'] > 0.1)
    node4 = int(row['EBIT/TA'] > 0.1)
    emb = [node1, node2, node3, node4]
    if node1 == 1:
        return 1, emb
    else:
        if node2 == 1:
            if node3 == 1:
                return 1, emb
            else:
                if node4 == 1:
                    return 1, emb
                else:
                    return 0, emb
        else:
            return 0, emb
