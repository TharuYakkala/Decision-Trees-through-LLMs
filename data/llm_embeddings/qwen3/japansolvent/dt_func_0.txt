def func(row):
    node1 = int(row['EBIT/TA'] <= 0.1)
    node2 = int(row['NI/TC'] <= 0.05)
    node3 = int(row['WC/TA'] <= 0.2)
    node4 = int(row['Equity/TL'] <= 0.5)
    node5 = int(row['Equity/TA'] <= 0.3)
    emb = [node1, node2, node3, node4, node5]
    
    if node1 == 1:
        prediction = 1
    elif node2 == 1:
        if node3 == 1:
            prediction = 0
        else:
            prediction = 1
    else:
        if node4 == 1:
            prediction = 1
        else:
            if node5 == 1:
                prediction = 1
            else:
                prediction = 0
    return prediction, emb
