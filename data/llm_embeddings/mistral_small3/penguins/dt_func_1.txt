def penguin_species_classifier(feature_row):
    # Precompute the binary vector embeddings
    node1 = int(feature_row['island'] == 2)  # Torgersen
    node2 = int(feature_row['island'] == 1)  # Dream
    node3 = int(feature_row['culmen_length_mm'] <= 42)
    node4 = int(feature_row['culmen_length_mm'] > 42)
    node5 = int(feature_row['culmen_depth_mm'] <= 16)
    node6 = int(feature_row['flipper_length_mm'] <= 190)
    node7 = int(feature_row['flipper_length_mm'] > 190)
    node8 = int(feature_row['body_mass_g'] <= 4000)
    node9 = int(feature_row['sex'] == 2)  # male

    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9]

    # Apply the decision tree logic
    if node1:  # Torgersen
        prediction = 2  # Gentoo
    elif node2:  # Dream
        if node3 and node5:  # culmen_length_mm <= 42 and culmen_depth_mm <= 16
            prediction = 0  # Adelie
        elif node6:  # flipper_length_mm <= 190
            prediction = 0  # Adelie
        else:  # flipper_length_mm > 190
            prediction = 2  # Gentoo
    else:  # Biscoe
        if node4 and node6:  # culmen_length_mm > 42 and flipper_length_mm <= 190
            prediction = 1  # Chinstrap
        elif node4 and node7:  # culmen_length_mm > 42 and flipper_length_mm > 190
            prediction = 2  # Gentoo
        elif node3 and node5:  # culmen_length_mm <= 42 and culmen_depth_mm <= 16
            prediction = 0  # Adelie
        elif node8:  # body_mass_g <= 4000
            prediction = 0  # Adelie
        else:  # body_mass_g > 4000
            prediction = 1  # Chinstrap

    return prediction, emb
