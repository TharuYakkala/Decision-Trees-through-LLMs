def predict_heart_disease(feature_row):
    node1 = int(feature_row['age'] > 50)
    node2 = int(feature_row['chest_pain'] == 0)
    node3 = int(feature_row['thalach'] < 150)
    node4 = int(feature_row['oldpeak'] > 2)
    node5 = int(feature_row['ca'] > 1)
    node6 = int(feature_row['thal'] == 1)

    emb = [node1, node2, node3, node4, node5, node6]

    if node1:
        if node2:
            if node3:
                if node4:
                    if node5:
                        if node6:
                            prediction = 1
                        else:
                            prediction = 0
                    else:
                        prediction = 0
                else:
                    prediction = 0
            else:
                prediction = 0
        else:
            prediction = 0
    else:
        prediction = 0

    return prediction, emb
