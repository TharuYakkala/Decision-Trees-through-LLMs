def heart_disease_classifier(feature_row):
    node1 = int(feature_row['oldpeak'] > 2)
    node2 = int(feature_row['slope'] == 3)
    node3 = int(feature_row['thalach'] < 100)
    node4 = int(feature_row['ca'] > 1)
    node5 = int(feature_row['thal'] == 1)
    node6 = int(feature_row['chest_pain'] == 0)
    node7 = int(feature_row['restecg'] == 3)
    node8 = int(feature_row['trestbps'] > 140)
    node9 = int(feature_row['exang'] == 2)
    node10 = int(feature_row['age'] > 50)
    node11 = int(feature_row['fbs'] == 2)
    node12 = int(feature_row['chol'] > 300)

    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9, node10, node11, node12]

    if node1 and node2 and not node3:
        return 1, emb
    elif node1 and not node2 and node4:
        return 1, emb
    elif node5 and node6 and node7:
        return 1, emb
    elif node8 and node9 and node10:
        return 1, emb
    elif node11 and node12 and node3:
        return 1, emb
    else:
        return 0, emb
