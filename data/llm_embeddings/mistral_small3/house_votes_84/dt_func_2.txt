def decision_tree_classifier(feature_row):
    # Precompute the binary vector embeddings
    node1 = int(feature_row['physician-fee-freeze'] == 2)
    node2 = int(feature_row['el-salvador-adi'] == 2)
    node3 = int(feature_row['anti-satellite-test-ban'] == 2)
    node4 = int(feature_row['crime'] == 2)
    node5 = int(feature_row['duty-free-exports'] == 2)
    node6 = int(feature_row['export-administration-act-south-africa'] == 2)

    emb = [node1, node2, node3, node4, node5, node6]

    # Decision tree logic
    if node1 and node2 and node3:
        prediction = 1
    elif node4 and node5:
        prediction = 0
    elif node6:
        prediction = 0
    else:
        prediction = 1

    return prediction, emb
