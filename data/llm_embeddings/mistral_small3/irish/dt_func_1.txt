def decision_tree(feature_row):
    node1 = int(feature_row['DVRT'] > 50)
    node2 = int(feature_row['Educational_level'] <= 3)
    node3 = int(feature_row['Prestige_score'] > 14)
    node4 = int(feature_row['Type_school'] == 2)
    node5 = int(feature_row['Sex'] == 1)

    emb = [node1, node2, node3, node4, node5]

    if node1 and node2:
        prediction = 1
    elif node1 and node3 and node4:
        prediction = 0
    elif node1 and node5:
        prediction = 1
    elif node2 and node3:
        prediction = 0
    else:
        prediction = 0

    return prediction, emb
