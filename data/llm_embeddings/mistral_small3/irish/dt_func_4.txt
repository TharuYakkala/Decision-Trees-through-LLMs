def decision_tree_predictor(feature_row):
    node1 = int(feature_row['DVRT'] >= 70)
    node2 = int(feature_row['Educational_level'] <= 5)
    node3 = int(feature_row['Prestige_score'] >= 20)
    node4 = int(feature_row['Type_school'] == 0)
    emb = [node1, node2, node3, node4]

    if not node1:
        if node2:
            if node3:
                prediction = 1
            else:
                prediction = 0
        else:
            if node4:
                prediction = 0
            else:
                prediction = 1
    else:
        if node2:
            prediction = 1
        else:
            prediction = 0

    return prediction, emb
