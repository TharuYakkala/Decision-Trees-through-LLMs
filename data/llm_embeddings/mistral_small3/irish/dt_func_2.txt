def decision_tree(feature_row):
    node1 = int(feature_row['DVRT'] > 100)
    node2 = int(feature_row['Prestige_score'] > 15)
    node3 = int(feature_row['Educational_level'] < 3)
    node4 = int(feature_row['Type_school'] == 1)
    node5 = int(feature_row['Sex'] == 1)

    emb = [node1, node2, node3, node4, node5]

    if node1 and node2:
        prediction = 1
    elif node3 and node4:
        prediction = 0
    elif node5 and not node3:
        prediction = 1
    else:
        prediction = 0

    return prediction, emb
