def decision_tree(row):
    node1 = int(row['DVRT'] > 50)
    node2 = int(row['Prestige_score'] > 15)
    node3 = int(row['Educational_level'] > 5)
    node4 = int(row['Type_school'] == 0)
    node5 = int(row['Sex'] == 1)

    emb = [node1, node2, node3, node4, node5]

    if node1 and node2:
        return 1, emb
    elif node3 and node4:
        return 1, emb
    elif node5:
        return 0, emb
    else:
        return 0, emb
