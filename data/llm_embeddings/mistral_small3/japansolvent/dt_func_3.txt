def func(row):
    # Precompute the binary vector embeddings
    node1 = int(row['Equity/TL'] < 0.5)
    node2 = int(row['EBIT/Sales'] > 0.1)
    node3 = int(row['WC/TA'] < 0.3)
    node4 = int(row['NI/Sales'] > 0.05)
    node5 = int(row['EBIT/TA'] > 0.08)
    node6 = int(row['Sales/TA'] < 0.8)
    node7 = int(row['Equity/TA'] > 0.4)
    node8 = int(row['NI/TC'] < 0.03)

    emb = [node1, node2, node3, node4, node5, node6, node7, node8]

    # Decision tree logic
    if node1 and node5:
        prediction = 1
    elif not node1 and node2 and node6:
        prediction = 1
    elif not node1 and node2 and not node6 and node4:
        prediction = 0
    elif not node1 and not node2 and node3 and node7:
        prediction = 1
    elif not node1 and not node2 and not node3 and node8:
        prediction = 0
    else:
        prediction = 0

    return prediction, emb
