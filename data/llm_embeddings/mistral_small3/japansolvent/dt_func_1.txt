def decision_tree(row):
    node1 = int(row['EBIT/TA'] > 0.1)
    node2 = int(row['NI/TC'] > 0.05)
    node3 = int(row['Sales/TA'] > 0.8)
    node4 = int(row['EBIT/Sales'] > 0.1)
    node5 = int(row['NI/Sales'] > 0.05)
    node6 = int(row['WC/TA'] > 0.2)
    node7 = int(row['Equity/TL'] > 0.5)
    node8 = int(row['Equity/TA'] > 0.4)

    emb = [node1, node2, node3, node4, node5, node6, node7, node8]

    if node1 and node2 and node3:
        prediction = 1
    elif node4 and node5 and node6 and node7:
        prediction = 1
    elif node8:
        prediction = 0
    else:
        prediction = 0

    return prediction, emb
