def survival_predictor(row):
    node1 = int(row['SPIDERS'] == 0)
    node2 = int(row['BILIRUBIN'] > 0.8)
    node3 = int(row['ALBUMIN'] < 2.5)
    node4 = int(row['ASCITES'] == 1)
    node5 = int(row['PROTIME'] > 10)
    node6 = int(row['AGE'] > 50)

    emb = [node1, node2, node3, node4, node5, node6]

    if node1 and node2 and node3:
        prediction = 1  # Survive
    elif node4 and node5 and node6:
        prediction = 0  # Die
    else:
        prediction = 1  # Survive

    return prediction, emb
