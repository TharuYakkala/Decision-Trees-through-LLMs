def decision_tree_row(row):
    # Precompute binary vector embeddings
    node1 = int(row['wage increase first year'] > 4.0)
    node2 = int(row['wage increase second year'] > 4.0)
    node3 = int(row['wage increase third year'] > 4.0)
    node4 = int(row['number of working hours during week'] < 38)
    node5 = int(row['shift differential'] > 15)
    node6 = int(row['education allowance'] == 'yes')
    node7 = int(row['number of statutory holidays'] > 12)
    node8 = int(row['longterm disability assistance'] == 'yes')
    node9 = int(row['contribution to health plan'] == 'full')
    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9]

    # Decision tree logic
    if node1 and node2 and node3:
        prediction = 1
    elif node4 and node5:
        prediction = 0
    elif node6 and node7 and node8 and node9:
        prediction = 1
    else:
        prediction = 0

    return prediction, emb
