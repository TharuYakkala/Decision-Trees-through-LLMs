def predict(row):
    """
    Decision tree classifier for company solvency.
    
    Parameters
    ----------
    row : dict
        A dictionary mapping feature names to their numeric values. 
        Expected keys:
            'EBIT/TA', 'Equity/TL', 'WC/TA', 'NI/Sales'
    
    Returns
    -------
    prediction : int
        1 if the company is predicted to be solvent, 0 otherwise.
    emb : list of int
        Binary vector of inner node evaluations.
    """
    # Inner node evaluations
    node0 = int(row['EBIT/TA'] > 0.02)          # EBIT/TA > 2%
    node1 = int(row['Equity/TL'] > 0.5)          # Equity/TL > 50%
    node2 = int(row['WC/TA'] > 0.05)             # WC/TA > 5%
    node3 = int(row['NI/Sales'] > 0.05)          # NI/Sales > 5%
    
    emb = [node0, node1, node2, node3]
    
    # Decision logic
    if node0 == 0:
        prediction = 0  # Insolvent
    else:
        if node1 == 1:
            if node2 == 1:
                prediction = 1  # Solvent
            else:
                prediction = 0  # Insolvent
        else:
            if node3 == 1:
                prediction = 1  # Solvent
            else:
                prediction = 0  # Insolvent
    
    return prediction, emb
