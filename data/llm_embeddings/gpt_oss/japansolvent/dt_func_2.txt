def predict_solvent(row):
    node1 = int(row['EBIT/TA'] > 0.15)
    node2 = int(row['Equity/TL'] > 0.6)
    node3 = int(row['NI/TC'] > 0.1)
    node4 = int(row['WC/TA'] > 0.1)
    emb = [node1, node2, node3, node4]
    if node1:
        if node2:
            pred = 1
        else:
            if node3:
                pred = 1
            else:
                pred = 0
    else:
        if node4:
            pred = 1
        else:
            pred = 0
    return pred, emb
