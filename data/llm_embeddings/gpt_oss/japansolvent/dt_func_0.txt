def predict(row):
    # compute binary embeddings for the internal nodes
    node1 = int(row['NI/TC'] > 0.08)          # Net Income / Total Capital
    node2 = int(row['Equity/TL'] > 0.4)       # Equity / Total Liabilities
    node3 = int(row['EBIT/TA'] > 0.05)        # EBIT / Total Assets
    node4 = int(row['Equity/TA'] > 0.25)      # Equity / Total Assets
    
    emb = [node1, node2, node3, node4]
    
    # decision tree logic
    if node1:          # NI/TC > 0.08
        if node2:      # Equity/TL > 0.4
            return 1, emb
        else:          # Equity/TL <= 0.4
            if node3:  # EBIT/TA > 0.05
                return 1, emb
            else:      # EBIT/TA <= 0.05
                return 0, emb
    else:              # NI/TC <= 0.08
        if node4:      # Equity/TA > 0.25
            return 1, emb
        else:          # Equity/TA <= 0.25
            return 0, emb
