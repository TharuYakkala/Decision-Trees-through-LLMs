def predict_solvent(row):
    # Binary decision nodes
    node1 = int(row['EBIT/TA'] > 0.02)          # EBIT relative to total assets
    node2 = int(row['Equity/TL'] > 0.5)         # Equity-to-liabilities ratio
    node3 = int(row['NI/Sales'] > 0.0)          # Net income relative to sales
    node4 = int(row['WC/TA'] > 0.1)             # Working capital relative to assets

    emb = [node1, node2, node3, node4]

    # Decision tree logic
    if node1:                 # EBIT/TA > 0.02
        if node2:            # Equity/TL > 0.5
            prediction = 1   # Solvent
        else:                # Equity/TL <= 0.5
            prediction = 1 if node3 else 0
    else:                     # EBIT/TA <= 0.02
        if node4:            # WC/TA > 0.1
            prediction = 1   # Solvent
        else:                # WC/TA <= 0.1
            prediction = 0   # Insolvent

    return prediction, emb
