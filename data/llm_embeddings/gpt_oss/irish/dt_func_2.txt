def predict(row):
    # Inner node evaluations
    node1 = int(row['Educational_level'] > 5)           # Is the student above the Junior Cycle terminal level?
    node2 = int(row['Prestige_score'] > 14)            # Is father's occupation prestige high?
    node3 = int(row['Type_school'] == 1)               # Is the school a secondary terminal leaver?
    node4 = int(row['Sex'] == 1)                      # Is the student male?

    # Binary vector of inner node truth values
    emb = [node1, node2, node3, node4]

    # Decision tree logic
    if node1 == 0:
        return 0, emb          # Not taken
    else:
        if node2 == 0:
            return 0, emb      # Not taken
        else:
            if node3 == 1:
                return 1, emb  # Taken
            else:
                if node4 == 1:
                    return 1, emb  # Taken
                else:
                    return 0, emb  # Not taken
