def decision_tree_classifier(row):
    # Precompute binary embeddings for inner nodes
    node_1 = int(row['Educational_level'] <= 5)
    node_2 = int(row['DVRT'] <= 20)
    node_3 = int(row['Prestige_score'] <= 10)
    node_4 = int(row['Sex'] == 1)           # 1 = male, 0 = female
    node_5 = int(row['Type_school'] == 0)   # 0 = primary_terminal_leaver
    emb = [node_1, node_2, node_3, node_4, node_5]

    # Decision tree logic
    if node_1:
        if node_2:
            if node_3:
                prediction = 0   # school leaving certificate not taken
            else:
                prediction = 1   # certificate taken
        else:
            prediction = 1
    else:
        if node_5:
            prediction = 0
        else:
            if node_4:
                prediction = 1
            else:
                prediction = 0

    return prediction, emb
