def predict_school_certificate(row):
    # Precompute binary embeddings for each decision node
    node1 = int(row['Educational_level'] <= 4)          # Low education level
    node2 = int(row['DVRT'] <= 75)                     # Low verbal reasoning score
    node3 = int(row['Prestige_score'] <= 15)           # Low prestige score

    emb = [node1, node2, node3]

    # Decision tree logic
    if node1:
        pred = 0   # Not taken
    else:
        if node2:
            pred = 0   # Not taken
        else:
            if node3:
                pred = 0   # Not taken
            else:
                pred = 1   # Taken

    return pred, emb
