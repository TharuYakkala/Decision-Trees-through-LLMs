def decision_tree_classifier(row):
    # Inner node evaluations
    node1 = int(row['Educational_level'] >= 8)          # Junior Cycle or higher?
    node2 = int(row['Type_school'] == 1)               # Secondary terminal leaver?
    node3 = int(row['DVRT'] >= 60)                     # Reasonable verbal reasoning score?
    
    # Binary embedding of node activations
    emb = [node1, node2, node3]
    
    # Decision tree logic
    if node1:
        prediction = 1
    else:
        if node2:
            prediction = 1
        else:
            if node3:
                prediction = 1
            else:
                prediction = 0
    
    return prediction, emb
