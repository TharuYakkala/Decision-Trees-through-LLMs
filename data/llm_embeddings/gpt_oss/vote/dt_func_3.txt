def predict_party(row):
    # Compute binary nodes
    node1 = int(row['el salvador aid'] == 2)          # 1 if yes, 0 if no
    node2 = int(row['mx missile'] == 2)              # 1 if yes, 0 if no
    node3 = int(row['education spending'] == 2)      # 1 if yes, 0 if no
    
    emb = [node1, node2, node3]
    
    # Decision tree logic
    if node1 == 1:               # El Salvador aid = yes → Democrat
        prediction = 0
    else:
        if node2 == 1:           # MX missile = yes → Republican
            prediction = 1
        else:
            if node3 == 1:       # Education spending = yes → Democrat
                prediction = 0
            else:                # Otherwise → Republican
                prediction = 1
    
    return prediction, emb
