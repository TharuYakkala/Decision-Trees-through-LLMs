def predict_heart(row):
    """
    Predicts whether a patient has heart disease (1) or not (0).
    Returns a tuple: (prediction, embedding_vector)
    """
    # Inner node evaluations
    node1 = int(row['age'] > 55)                         # age > 55
    node2 = int(row['chest_pain'] == 0)                 # chest pain type = 0 (asymptomatic)
    node3 = int(row['oldpeak'] > 1.5)                   # ST depression > 1.5
    node4 = int(row['sex'] == 'male')                  # male
    node5 = int(row['fbs'] == 2)                       # fasting blood sugar true
    node6 = int(row['trestbps'] > 140)                 # resting BP > 140
    node7 = int(row['chol'] > 240)                     # cholesterol > 240

    emb = [node1, node2, node3, node4, node5, node6, node7]

    # Decision tree logic
    if node1 == 0:   # age <= 55
        if node2 == 1:   # asymptomatic
            pred = 0
        else:           # not asymptomatic
            if node3 == 1:   # oldpeak > 1.5
                pred = 1
            else:
                pred = 0
    else:           # age > 55
        if node4 == 0:   # female
            if node5 == 1:   # fbs true
                pred = 1
            else:
                pred = 0
        else:           # male
            if node6 == 1:   # trestbps > 140
                pred = 1
            else:
                if node7 == 1:   # chol > 240
                    pred = 1
                else:
                    pred = 0

    return pred, emb
