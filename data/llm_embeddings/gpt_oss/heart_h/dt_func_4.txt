def heart_disease_classifier(row):
    # Precompute binary embeddings for inner nodes
    node1 = int(row['chest_pain'] == 3)            # Typical angina
    node2 = int(row['age'] > 55)                  # Older age
    node3 = int(row['trestbps'] > 140)            # High resting blood pressure
    node4 = int(row['chol'] > 240)                # High cholesterol
    node5 = int(row['fbs'] == 2)                  # Fasting blood sugar > 120 mg/dl (true)
    emb = [node1, node2, node3, node4, node5]

    # Decision tree logic
    if node1:
        prediction = 1
    else:
        if node2:
            prediction = 1
        else:
            if node3:
                prediction = 1
            else:
                if node4:
                    prediction = 1
                else:
                    if node5:
                        prediction = 1
                    else:
                        prediction = 0
    return prediction, emb
