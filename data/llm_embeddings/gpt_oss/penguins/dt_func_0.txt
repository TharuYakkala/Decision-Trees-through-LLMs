    # Precompute binary decision embeddings
    node1 = int(row['flipper_length_mm'] <= 190)          # flipper <= 190 mm ?

    node3 = int(row['body_mass_g'] <= 3800)              # body mass <= 3800 g ?
    emb = [node1, node2, node3]

    # Decision tree logic
    if node1 == 1:                       # flipper_length_mm <= 190
        if node2 == 1:                   # body_mass_g <= 3500
            pred = 0                    # Adelie
        else:                            # body_mass_g > 3500
            pred = 1                    # Chinstrap
    else:                                # flipper_length_mm > 190
        if node3 == 1:                   # body_mass_g <= 3800
            pred = 1                    # Chinstrap
