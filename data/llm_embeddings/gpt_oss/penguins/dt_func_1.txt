def predict_penguin(row):
    # Precompute binary conditions for inner nodes
    node1 = int(row['body_mass_g'] <= 3800)          # Is body mass light enough for Adelie/Chinstrap?
    node2 = int(row['culmen_length_mm'] <= 19)      # Is culmen short enough for Adelie?
    node3 = int(row['flipper_length_mm'] <= 190)    # Is flipper short enough for Gentoo?
    
    emb = [node1, node2, node3]                     # Binary vector of node evaluations
    
    # Decision tree logic
    if node1 == 1:                                 # Light-bodied birds
        if node2 == 1:
            pred = 0          # Adelie
        else:
            pred = 1          # Chinstrap
    else:                                          # Heavy-bodied birds
        if node3 == 1:
            pred = 2          # Gentoo
        else:
            pred = 1          # Chinstrap
    
    return pred, emb
