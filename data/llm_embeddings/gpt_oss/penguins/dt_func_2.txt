def penguin_species_classifier(row):
    # Precompute binary node evaluations
    node1 = int(row['body_mass_g'] <= 3800)          # Node 1: light-bodied species?
    node2 = int(row['culmen_length_mm'] <= 23.5)     # Node 2: short culmen
    node3 = int(row['flipper_length_mm'] <= 215)     # Node 3: short flipper
    emb = [node1, node2, node3]

    # Decision tree logic
    if node1:
        if node2:
            pred = 0  # Adelie
        else:
            pred = 1  # Chinstrap
    else:
        if node3:
            pred = 2  # Gentoo
        else:
            pred = 1  # Chinstrap

    return pred, emb
