def penguin_predict(row):
    """
    Predicts penguin species (0=Adelie, 1=Chinstrap, 2=Gentoo)
    and returns the binary truth vector for the decision nodes.
    """
    # Decision nodes
    node1 = int(row['island'] == 2)                     # Torgersen island
    node2 = int(row['body_mass_g'] > 3800)              # heavy body mass
    node3 = int(row['flipper_length_mm'] > 200)         # long flipper

    emb = [node1, node2, node3]

    # Tree logic
    if node1:               # Torgersen → Gentoo
        pred = 2
    else:
        if node2:           # heavy mass → Gentoo
            pred = 2
        else:
            if node3:       # long flipper → Chinstrap
                pred = 1
            else:           # otherwise Adelie
                pred = 0

    return pred, emb
