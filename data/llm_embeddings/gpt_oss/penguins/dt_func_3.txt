def penguin_species(row):
    """
    Predicts penguin species (0 = Adelie, 1 = Chinstrap, 2 = Gentoo)
    and returns the binary truth vector of the inner decision nodes.

    Parameters
    ----------
    row : dict or pandas.Series
        A mapping from feature name to its value.

    Returns
    -------
    prediction : int
        Predicted class label.
    emb : list[int]
        Truth values of the inner nodes in the order they are evaluated.
    """

    # Inner node evaluations
    node1 = int(row['flipper_length_mm'] <= 215)           # Adelie vs others
    node2 = int(row['flipper_length_mm'] > 215)            # Complement of node1
    node3 = int(row['body_mass_g'] <= 3600)                # Chinstrap vs Gentoo
    node4 = int(row['body_mass_g'] > 3600)                 # Complement of node3

    emb = [node1, node2, node3, node4]

    # Decision logic
    if node1:
        prediction = 0          # Adelie
    elif node2 and node3:
        prediction = 1          # Chinstrap
    else:                       # node2 and node4
        prediction = 2          # Gentoo

    return prediction, emb
