def predict_party(row):
    # Precompute binary embeddings for the decision tree nodes
    node1 = int(row['crime'] == 2)                     # Node 1: Did the legislator vote 'yes' on crime?
    node2 = int(row['education-spending'] == 2)       # Node 2: Did the legislator vote 'yes' on education spending?

    emb = [node1, node2]

    # Decision tree logic
    if node1 == 1:           # Voted yes on crime -> Republican
        prediction = 1
    else:                    # Voted no on crime
        if node2 == 1:       # Voted yes on education spending -> Democrat
            prediction = 0
        else:                # Voted no on education spending -> Republican
            prediction = 1

    return prediction, emb
