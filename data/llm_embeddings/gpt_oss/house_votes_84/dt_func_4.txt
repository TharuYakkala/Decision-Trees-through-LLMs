def predict_party(row):
    """
    Predict party affiliation based on a simple decision tree.
    Returns a tuple (prediction, emb) where:
        prediction : 1 for Republican, 0 for Democrat
        emb        : list of binary evaluations of internal nodes
    """
    # Binary evaluation of internal nodes
    node1 = int(row['handicapped-infants'] == 2)                     # Yes on handicapped infants
    node2 = int(row['immigration'] == 2)                             # Yes on immigration
    node3 = int(row['water-project-cost-sharing'] == 2)              # Yes on water project cost sharing
    node4 = int(row['crime'] == 2)                                   # Yes on crime
    node5 = int(row['export-administration-act-south-africa'] == 2) # Yes on SA export act

    emb = [node1, node2, node3, node4, node5]

    # Decision tree logic
    if node1 == 1:          # Handicapped infants yes
        if node2 == 1:      # Immigration yes
            pred = 1       # Republican
        else:
            if node4 == 1:  # Crime yes
                pred = 1   # Republican
            else:
                pred = 0   # Democrat
    else:                   # Handicapped infants no
        if node3 == 1:      # Water project cost sharing yes
            pred = 1       # Republican
        else:
            pred = 0       # Democrat

    return pred, emb
