def predict_party(row):
    # Precompute binary conditions
    node1 = int(row['handicapped-infants'] == 2)          # Yes => Republican
    node2 = int(row['immigration'] == 2)                 # Yes => Republican
    node3 = int(row['duty-free-exports'] == 2)           # Yes => Republican
    
    emb = [node1, node2, node3]
    
    # Decision logic
    if node1 == 1:
        prediction = 1  # Republican
    elif node2 == 1:
        prediction = 1
    elif node3 == 1:
        prediction = 1
    else:
        prediction = 0  # Democrat
    
    return prediction, emb
