def predict_survival(row):
    # Pre‑compute binary node evaluations
    node1 = int(row['BILIRUBIN'] <= 1.5)      # Node 1: bilirubin low
    node2 = int(row['ASCITES'] == 1)         # Node 2: ascites present
    node3 = int(row['ALBUMIN'] < 2.0)         # Node 3: albumin low
    emb = [node1, node2, node3]

    # Decision tree logic
    if node1 == 1:
        pred = 1          # Survives
    else:
        if node2 == 1:
            pred = 0      # Dies
        else:
            if node3 == 1:
                pred = 0  # Dies
            else:
                pred = 1  # Survives

    return pred, emb
