def predict_survival(row):
    # Node evaluations
    node1 = int(row['PROTIME'] <= 20)          # Prothrombin time <= 20
    node2 = int(row['ALBUMIN'] <= 2.5)        # Albumin <= 2.5
    node3 = int(row['BILIRUBIN'] > 3.0)       # Bilirubin > 3.0
    node4 = int(row['ASCITES'] == 1)          # Ascites present
    node5 = int(row['VARICES'] == 1)          # Varices present
    
    emb = [node1, node2, node3, node4, node5]
    
    # Decision tree logic
    if node1:
        if node2:
            prediction = 0  # Die
        else:
            if node3:
                prediction = 0
            else:
                prediction = 1  # Live
    else:
        if node4:
            prediction = 0
        else:
            if node5:
                prediction = 0
            else:
                prediction = 1
    
    return prediction, emb
