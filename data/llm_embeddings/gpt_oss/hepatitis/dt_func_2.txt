def predict(row):
    # Precompute binary node evaluations
    node1 = int(row['BILIRUBIN'] <= 3)          # Bilirubin low
    node2 = int(row['ALBUMIN'] >= 2.5)          # Albumin normal or high
    node3 = int(row['PROTIME'] <= 18)           # Prothrombin time normal or low
    node4 = int(row['AGE'] <= 70)               # Age not too old
    emb = [node1, node2, node3, node4]
    
    # Decision tree logic
    if node1:
        if node2:
            if node3:
                if node4:
                    pred = 1   # patient survives
                else:
                    pred = 0   # patient dies (older age)
            else:
                pred = 0       # abnormal protime
        else:
            pred = 0           # low albumin
    else:
        pred = 0               # high bilirubin
    
    return pred, emb
