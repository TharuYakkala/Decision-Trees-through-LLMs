def predict_labor_outcome(row):
    """
    Predicts whether the outcome of labor negotiations is good (1) or bad (0)
    based on selected key features of the agreement.

    Parameters
    ----------
    row : dict or pandas.Series
        A single record containing the following feature keys:
            'wage increase first year'          : float
            'cost of living adjustment'         : str ('none', 'tcf', 'tc')
            'pension'                           : str ('none', 'ret_allw', 'empl_contr')
            'standby pay'                       : float

    Returns
    -------
    prediction : int
        1 for a good outcome, 0 for a bad outcome.
    emb : list of int
        Binary vector indicating the truth value (1) or falsity (0) of each
        inner node condition in the decision tree.
    """
    # Pre‑compute binary node embeddings
    node_1 = int(row['wage increase first year'] > 4.5)
    node_2 = int(row['cost of living adjustment'] != 'none')
    node_3 = int(row['pension'] != 'none')
    node_4 = int(row['standby pay'] > 10)

    emb = [node_1, node_2, node_3, node_4]

    # Decision tree logic
    if node_1:                     # wage increase first year > 4.5
        if node_2:                 # cost of living adjustment present
            prediction = 1
        else:                      # no cost of living adjustment
            if node_3:             # pension contribution present
                prediction = 1
            else:                  # no pension contribution
                prediction = 0
    else:                           # wage increase first year <= 4.5
        if node_3:                 # pension contribution present
            if node_4:             # standby pay > 10
                prediction = 1
            else:                  # standby pay <= 10
                prediction = 0
        else:                      # no pension contribution
            prediction = 0

    return prediction, emb
