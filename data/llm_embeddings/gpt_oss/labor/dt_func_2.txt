def classify(row):
    node1 = int(row['cost of living adjustment'] == 'none')
    node2 = int(row['pension'] == 'none')
    node3 = int(row['education allowance'] == True)
    node4 = int(row['wage increase first year'] > 4.5)
    node5 = int(row['standby pay'] > 15)
    emb = [node1, node2, node3, node4, node5]

    if node1 == 1:
        if node2 == 1:
            pred = 0
        else:
            if node3 == 1:
                pred = 1
            else:
                pred = 0
    else:
        if node4 == 1:
            pred = 1
        else:
            if node5 == 1:
                pred = 1
            else:
                pred = 0

    return pred, emb
