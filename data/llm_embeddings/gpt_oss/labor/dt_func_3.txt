def predict_labor_negotiation(row):
    # Precompute binary node values
    node1 = int(row['duration'] > 4)                                 # duration > 4?
    node2 = int(row['pension'] != 'none')                            # pension present?
    node3 = int(row['cost of living adjustment'] != 'none')          # cost of living adjustment present?
    node4 = int(row['wage increase first year'] >= 5.5)              # first year wage increase high?
    node5 = int(row['shift differential'] > 10)                      # shift differential > 10?
    node6 = int(row['education allowance'] == 'true')                # education allowance provided?
    node7 = int(row['statutory holidays'] >= 12)                     # statutory holidays >= 12?

    emb = [node1, node2, node3, node4, node5, node6, node7]

    # Decision tree logic
    if node1:
        if node2:
            prediction = 1
        else:
            if node3:
                prediction = 1
            else:
                prediction = 0
    else:
        if node4:
            if node5:
                prediction = 1
            else:
                if node6:
                    prediction = 1
                else:
                    prediction = 0
        else:
            if node7:
                prediction = 1
            else:
                prediction = 0

    return prediction, emb
