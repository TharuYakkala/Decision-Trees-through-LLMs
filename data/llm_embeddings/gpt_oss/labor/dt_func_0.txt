def predict(row):
    # Precompute binary node embeddings
    node1 = int(row['wage increase first year'] >= 5.0)  # 1 if first year increase is high
    node2 = int(row['cost of living adjustment'] != 'none')  # 1 if a COLA is offered
    node3 = int(row['pension'] != 'none')  # 1 if pension contributions exist
    node4 = int(row['wage increase second year'] >= 5.0)  # 1 if second year increase is high
    node5 = int(row['vacation'] == 'gnr')  # 1 if vacation days are generous
    
    emb = [node1, node2, node3, node4, node5]
    
    # Decision tree logic
    if node1:  # high first-year wage increase
        if node2:  # cost-of-living adjustment present
            prediction = 1  # good outcome
        else:
            if node3:  # pension contributions present
                prediction = 1
            else:
                prediction = 0  # bad outcome
    else:  # low first-year wage increase
        if node4:  # high second-year wage increase
            prediction = 1
        else:
            if node5:  # generous vacation days
                prediction = 1
            else:
                prediction = 0
    return prediction, emb
