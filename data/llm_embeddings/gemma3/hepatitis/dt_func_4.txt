def predict_survival(row):
    """
    Predicts patient survival based on the given features.

    Args:
        row (dict): A dictionary containing the feature values for a single patient.

    Returns:
        tuple: A tuple containing the predicted survival (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['AGE'] > 60)
    node2 = int(row['SEX'] == 2)  # Male
    node3 = int(row['STEROIDS'] == 1)
    node4 = int(row['FATIGUE'] == 1)
    node5 = int(row['BILIRUBIN'] > 2.0)
    node6 = int(row['ALBUMIN'] < 3.0)
    node7 = int(row['PROTIME'] > 40)
    node8 = int(row['HISTOLOGY'] == 1)
    
    emb = [node1, node2, node3, node4, node5, node6, node7, node8]

    if node1 == 0:
        if node2 == 1:
            if node3 == 1:
                return 0, emb
            else:
                return 1, emb
        else:
            if node4 == 1:
                if node5 == 1:
                    return 0, emb
                else:
                    return 1, emb
            else:
                if node6 == 1:
                    if node7 == 1:
                        return 0, emb
                    else:
                        return 1, emb
                else:
                    if node8 == 1:
                        return 1, emb
                    else:
                        return 1, emb
    else:
        if node5 == 1:
            if node6 == 1:
                return 0, emb
            else:
                return 1, emb
        else:
            return 1, emb
