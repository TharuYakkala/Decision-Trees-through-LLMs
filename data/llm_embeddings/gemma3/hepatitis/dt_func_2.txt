def predict_survival(row):
    """
    Predicts patient survival (0 = die, 1 = live) based on a decision tree.

    Args:
        row (dict): A dictionary containing the feature values for a patient.

    Returns:
        tuple: A tuple containing the predicted survival (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['AGE'] > 50)
    node2 = int(row['SEX'] == 1)  # Female
    node3 = int(row['STEROID'] == 1)
    node4 = int(row['ANTIVIRALS'] == 1)
    node5 = int(row['BILIRUBIN'] > 2.0)
    node6 = int(row['ALK PHOSPHATE'] > 100)
    node7 = int(row['SGOT'] > 50)
    node8 = int(row['PROTIME'] > 30)

    emb = [node1, node2, node3, node4, node5, node6, node7, node8]

    if node1 == 0:
        if node2 == 1:
            return 1, emb
        else:
            if node3 == 1:
                if node4 == 1:
                    return 1, emb
                else:
                    if node5 == 0:
                        return 0, emb
                    else:
                        return 0, emb
            else:
                if node5 == 0:
                    if node6 == 0:
                        return 1, emb
                    else:
                        if node7 == 0:
                            return 1, emb
                        else:
                            return 0, emb
                else:
                    if node8 == 0:
                        return 0, emb
                    else:
                        return 0, emb
    else:
        if node5 == 0:
            if node6 == 0:
                if node7 == 0:
                    return 1, emb
                else:
                    return 0, emb
            else:
                return 0, emb
        else:
            return 0, emb
