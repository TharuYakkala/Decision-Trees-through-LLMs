def penguin_species_classifier(row):
    """
    Classifies penguin species based on given features.

    Args:
        row (dict): A dictionary containing penguin feature values.

    Returns:
        tuple: A tuple containing the predicted species (0, 1, or 2) 
               and a list of binary node embeddings.
    """

    # Precompute binary embeddings
    node1 = int(row['island'] == 0)  # Biscoe
    node2 = int(row['culmen_length_mm'] > 35)
    node3 = int(row['culmen_depth_mm'] > 17)
    node4 = int(row['flipper_length_mm'] > 200)
    node5 = int(row['body_mass_g'] > 4000)
    node6 = int(row['sex'] == 2)  # Male

    emb = [node1, node2, node3, node4, node5, node6]

    # Decision Tree Logic
    if node1 == 1:  # Biscoe
        return 0, emb
    elif node2 == 1 and node3 == 1:  # culmen_length > 35 and culmen_depth > 17
        return 2, emb
    elif node4 == 1 and node5 == 1: # flipper_length > 200 and body_mass > 4000
        return 2, emb
    elif node6 == 1: # male
        return 1, emb
    else:  # Default to Adelie
        return 0, emb
