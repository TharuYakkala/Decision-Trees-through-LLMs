def penguin_species_classifier(row):
    """
    Classifies penguin species based on a decision tree.

    Args:
        row (dict): A dictionary containing penguin features.

    Returns:
        tuple: A tuple containing the predicted species (0, 1, or 2) 
               and a list representing the truth values of the inner nodes.
    """

    # Precompute binary vector embeddings
    node1 = int(row['island'] == 0)  # Biscoe
    node2 = int(row['culmen_length_mm'] > 34)
    node3 = int(row['culmen_depth_mm'] > 17)
    node4 = int(row['flipper_length_mm'] > 190)
    node5 = int(row['body_mass_g'] > 3500)
    node6 = int(row['sex'] == 2) # Male

    emb = [node1, node2, node3, node4, node5, node6]

    # Decision Tree Logic
    if node1 == 1: # Island = Biscoe
        return 0, emb
    elif node2 == 1: # culmen_length_mm > 34
        if node3 == 1: # culmen_depth_mm > 17
            return 2, emb
        else:
            return 1, emb
    else: # culmen_length_mm <= 34
        if node4 == 1: # flipper_length_mm > 190
            if node5 == 1: # body_mass_g > 3500
                return 2, emb
            else:
                return 1, emb
        else:
            if node6 == 1:  # sex == male
                return 1, emb
            else:
                return 0, emb
