def predict_penguin_species(row):
    """
    Predicts penguin species based on the given features.

    Args:
        row (dict): A dictionary containing the penguin features.

    Returns:
        tuple: A tuple containing the predicted species (0, 1, or 2) and a list of binary node embeddings.
    """

    node1 = int(row['island'] == 0)  # Biscoe
    node2 = int(row['culmen_length_mm'] > 35)
    node3 = int(row['culmen_depth_mm'] > 17)
    node4 = int(row['flipper_length_mm'] > 200)
    node5 = int(row['body_mass_g'] > 4000)
    node6 = int(row['sex'] == 2) # Male

    emb = [node1, node2, node3, node4, node5, node6]

    if node1 == 0:  # Biscoe
        if node2 == 0: # culmen_length_mm <= 35
            return 0, emb
        else: # culmen_length_mm > 35
            if node3 == 0:  # culmen_depth_mm <= 17
                return 1, emb
            else: # culmen_depth_mm > 17
                return 2, emb
    else: # Dream or Torgersen
        if node4 == 0: # flipper_length_mm <= 200
            if node5 == 0: # body_mass_g <= 4000
                return 1, emb
            else: # body_mass_g > 4000
                return 2, emb
        else: # flipper_length_mm > 200
            if node6 == 0: # sex == 0 or 1
                return 2, emb
            else: # sex == 2
                return 2, emb
