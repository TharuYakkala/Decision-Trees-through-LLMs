def predict_school_leaving_certificate(row):
    """
    Predicts whether a school leaving certificate is taken based on the given features.

    Args:
        row (dict): A dictionary containing the feature values for a single observation.

    Returns:
        tuple: A tuple containing the prediction (0 or 1) and a list representing the truth values of the inner nodes.
    """

    node1 = int(row['Sex'] == 1)
    node2 = int(row['DVRT'] > 60)
    node3 = int(row['Educational_level'] > 4)
    node4 = int(row['Prestige_score'] > 14)
    node5 = int(row['Type_school'] == 1)
    
    emb = [node1, node2, node3, node4, node5]

    if node1 == 0:  # Female
        if node2 == 1:  # DVRT > 60
            return 1, emb
        else:
            if node3 == 1: # Educational level > 4
                return 1, emb
            else:
                return 0, emb
    else:  # Male
        if node2 == 1:  # DVRT > 60
            if node4 == 1: #Prestige score > 14
                return 1, emb
            else:
                if node5 == 1: #Type of school == 1
                    return 1, emb
                else:
                    return 0, emb
        else:
            if node3 == 1: # Educational level > 4
                if node5 == 1: #Type of school == 1
                    return 1, emb
                else:
                    return 0, emb
            else:
                return 0, emb
