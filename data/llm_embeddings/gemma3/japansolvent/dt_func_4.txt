def predict_solvency(row):
    """
    Predicts whether a company is solvent based on financial ratios.

    Args:
        row (dict): A dictionary containing the feature values.

    Returns:
        tuple: A tuple containing the predicted class (0 or 1) and a list of binary embeddings.
    """

    node1 = int(row['EBIT/TA'] > 0.05)
    node2 = int(row['NI/TC'] > 0.10)
    node3 = int(row['Sales/TA'] > 1.0)
    node4 = int(row['EBIT/Sales'] > 0.25)
    node5 = int(row['NI/Sales'] > 0.05)
    node6 = int(row['WC/TA'] > 0.15)
    node7 = int(row['Equity/TL'] > 0.5)
    node8 = int(row['Equity/TA'] > 0.3)

    emb = [node1, node2, node3, node4, node5, node6, node7, node8]

    if node1 == 0:
        return 0, emb
    elif node2 == 0:
        return 0, emb
    elif node3 == 0:
        return 0, emb
    elif node4 == 0:
        return 0, emb
    elif node5 == 0:
        return 0, emb
    elif node6 == 0:
        return 0, emb
    elif node7 == 0:
        return 0, emb
    elif node8 == 0:
        return 0, emb
    else:
        return 1, emb
