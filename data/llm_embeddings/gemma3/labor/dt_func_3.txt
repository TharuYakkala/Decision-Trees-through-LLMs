def negotiate_outcome(row):
    """
    Predicts the outcome of labor negotiations based on a decision tree.

    Args:
        row (dict): A dictionary containing the feature values.

    Returns:
        tuple: A tuple containing the predicted class (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['wage increase first year'] <= 3.0)
    node2 = int(row['cost of living adjustment'] != 'none')
    node3 = int(row['working hours'] > 38)
    node4 = int(row['standby pay'] > 10)
    node5 = int(row['education allowance'] == 'true')
    node6 = int(row['statutory holidays'] >= 12)
    node7 = int(row['vacation'] == 'generous')
    node8 = int(row['longterm disability assistance'] == 'true')
    node9 = int(row['contribution to dental plan'] == 'full')
    node10 = int(row['bereavement assistance'] == 'true')
    node11 = int(row['contribution to health plan'] == 'full')

    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9, node10, node11]

    if node1 == 0:
        if node2 == 0:
            return 0, emb
        else:
            if node3 == 0:
                return 0, emb
            else:
                if node4 == 0:
                    return 0, emb
                else:
                    return 1, emb
    else:
        if node5 == 0:
            if node6 == 0:
                return 1, emb
            else:
                if node7 == 0:
                    return 1, emb
                else:
                    return 1, emb
        else:
            if node8 == 0:
                if node9 == 0:
                    return 1, emb
                else:
                    if node10 == 0:
                        return 1, emb
                    else:
                        return 1, emb
            else:
                if node11 == 0:
                    return 1, emb
                else:
                    return 1, emb
