def negotiate_outcome(row):
    """
    Predicts the outcome of labor negotiations based on the provided features.

    Args:
        row (dict): A dictionary containing the feature values.

    Returns:
        tuple: A tuple containing the predicted outcome (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['wage increase first year'] <= 3.0)
    node2 = int(row['cost of living adjustment'] != 'none')
    node3 = int(row['working hours'] > 37)
    node4 = int(row['pension'] != 'none')
    node5 = int(row['shift differential'] > 10)
    node6 = int(row['statutory holidays'] >= 12)
    node7 = int(row['vacation'] == 'gnr')
    node8 = int(row['longterm disability assistance'] == 'true')
    node9 = int(row['contribution to dental plan'] == 'full')
    node10 = int(row['contribution to health plan'] == 'full')


    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9, node10]

    if node1 == 1 and node2 == 1:
        return 1, emb
    elif node1 == 0 and node3 == 1 and node4 == 1:
        return 1, emb
    elif node1 == 0 and node3 == 0 and node5 == 1 and node6 == 1:
        return 1, emb
    elif node1 == 0 and node5 == 0 and node7 == 1 and node8 == 1:
        return 1, emb
    elif node9 == 1 and node10 == 1:
        return 1, emb
    else:
        return 0, emb
