def predict_labor_negotiations(row):
    """
    Predicts the outcome of labor negotiations based on a decision tree.

    Args:
        row (dict): A dictionary containing the feature values.

    Returns:
        tuple: A tuple containing the prediction (0 or 1) and a list of binary node embeddings.
    """

    # Precompute binary embeddings
    node1 = int(row['duration'] <= 3)
    node2 = int(row['wage increase first year'] <= 3)
    node3 = int(row['cost of living adjustment'] != 'none')
    node4 = int(row['working hours'] > 38)
    node5 = int(row['pension'] != 'none')
    node6 = int(row['standby pay'] > 10)
    node7 = int(row['shift differential'] > 10)
    node8 = int(row['education allowance'] == 'true')
    node9 = int(row['statutory holidays'] >= 12)
    node10 = int(row['vacation'] == 'ba')
    node11 = int(row['longterm disability assistance'] == 'true')
    node12 = int(row['contribution to dental plan'] == 'full')
    node13 = int(row['bereavement assistance'] == 'true')
    node14 = int(row['contribution to health plan'] == 'full')

    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9, node10, node11, node12, node13, node14]

    # Decision Tree Logic
    if node1 == 0:
        return 0, emb
    elif node2 == 1:
        return 0, emb
    elif node3 == 1:
        if node5 == 1:
            return 1, emb
        else:
            return 0, emb
    elif node4 == 1:
        if node6 == 1:
            return 1, emb
        else:
            return 0, emb
    elif node7 == 1:
        return 1, emb
    elif node8 == 1:
        if node9 == 1:
            return 1, emb
        else:
            return 0, emb
    elif node10 == 1:
        return 1, emb
    elif node11 == 1:
        return 1, emb
    elif node12 == 1:
        return 1, emb
    elif node13 == 1:
        return 1, emb
    elif node14 == 1:
        return 1, emb
    else:
        return 0, emb
