def func(row):
    node1 = int(row['rectal_temperature'] <= 38.5)
    node2 = int(row['abdominal_distension'] <= 2)
    node3 = int(row['pulse'] <= 60)
    emb = [node1, node2, node3]
    if node1 == 1:
        if node2 == 1:
            prediction = 0
        else:
            prediction = 1
    else:
        if node3 == 1:
            prediction = 0
        else:
            prediction = 1
    return prediction, emb
