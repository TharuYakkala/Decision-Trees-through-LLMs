def func(row):
    node1 = int(row['oldpeak'] > 2.5)
    node2 = int(row['thalach'] < 130)
    node3 = int(row['exang'] == 2)
    emb = [node1, node2, node3]
    if node1 == 0:
        prediction = 0
    else:
        if node2 == 1:
            prediction = 1
        else:
            prediction = 0
    return prediction, emb
