def func(row):
    node1 = int(row['Derogatory.reports'] > 2)
    node2 = int(row['Own.home'] == 1)
    node3 = int(row['Income.per.dependent'] > 5)
    emb = [node1, node2, node3]
    if node1 == 1:
        return 0, emb
    else:
        if node2 == 1:
            return 1, emb
        else:
            if node3 == 1:
                return 1, emb
            else:
                return 0, emb
