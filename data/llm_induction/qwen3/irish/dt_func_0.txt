def func(row):
    node1 = int(row['DVRT'] <= 20)
    node2 = int(row['Prestige_score'] <= 15)
    node3 = int(row['Prestige_score'] <= 20)
    emb = [node1, node2, node3]
    if node1 == 1:
        if node2 == 1:
            prediction = 0
        else:
            prediction = 1
    else:
        if node3 == 1:
            prediction = 1
        else:
            prediction = 0
    return prediction, emb
