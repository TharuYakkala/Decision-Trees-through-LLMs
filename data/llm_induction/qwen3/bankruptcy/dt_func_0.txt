def func(row):
    node1 = int(row['BVE/BVL'] <= 1.0)
    node2 = int(row['EBIT/TA'] <= 0.05)
    node3 = int(row['WC/TA'] <= -0.10)
    emb = [node1, node2, node3]
    if node1 == 1:
        if node2 == 1:
            prediction = 1
        else:
            prediction = 0
    else:
        if node3 == 1:
            prediction = 1
        else:
            prediction = 0
    return prediction, emb
