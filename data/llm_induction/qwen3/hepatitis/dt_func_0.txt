def func(row):
    node1 = int(row['BILIRUBIN'] > 2.0)
    node2 = int(row['ALBUMIN'] < 3.0)
    node3 = int(row['PROTIME'] > 30)
    emb = [node1, node2, node3]
    if node1 == 1:
        prediction = 0
    else:
        if node2 == 1:
            prediction = 0
        else:
            prediction = 1
    return prediction, emb
