def predict(row):
    # Precompute binary node evaluations
    node1 = int(row['rectal_temperature'] > 39.0)          # Root node
    node2 = int(row['pulse'] > 120)                       # Left child split
    node3 = int(row['peripheral_pulse'] == 1)             # Right child split

    emb = [node1, node2, node3]

    # Decision tree logic (depth 2)
    if node1 == 1:          # rectal_temperature > 39.0
        if node2 == 1:      # pulse > 120
            prediction = 1  # surgery
        else:
            prediction = 0  # no surgery
    else:                   # rectal_temperature <= 39.0
        if node3 == 1:      # peripheral_pulse == 1 (absent)
            prediction = 1  # surgery
        else:
            prediction = 0  # no surgery

    return prediction, emb
