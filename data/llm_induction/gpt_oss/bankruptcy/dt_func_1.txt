def predict(row):
    # Precompute binary node evaluations
    node1 = int(row['BVE/BVL'] <= 0.3)          # Root node: leverage threshold
    node2 = int(row['WC/TA'] <= 0.05)           # Child node: liquidity threshold
    emb = [node1, node2]

    # Decision tree logic
    if node1 == 1:
        pred = 1  # High leverage → bankruptcy
    else:
        if node2 == 1:
            pred = 1  # Low liquidity → bankruptcy
        else:
            pred = 0  # Otherwise → not bankrupt

    return pred, emb
