def predict(row):
    # Precompute binary embeddings for the inner nodes
    node1 = int(row['Book Value of Equity/Book Value of Liabilities'] <= 0.5)
    node2 = int(row['Working Capital/Total Assets'] <= 0.1)
    emb = [node1, node2]

    # Decision tree logic
    if node1 == 1:
        if node2 == 1:
            pred = 0  # not bankrupt
        else:
            pred = 1  # bankrupt
    else:
        pred = 1      # bankrupt

    return pred, emb
