def predict_party(row):
    # Inner node evaluations
    node1 = int(row['handicapped-infants'] == 2)          # Vote on handicapped infants
    node2 = int(row['water-project-cost-sharing'] == 2)   # Vote on water project cost sharing
    emb = [node1, node2]

    # Decision tree logic (depth 2)
    if node1:          # handicapped-infants == yes
        if node2:      # water-project-cost-sharing == yes
            pred = 1   # Republican
        else:
            pred = 0   # Democrat
    else:              # handicapped-infants == no
        pred = 0       # Democrat

    return pred, emb
