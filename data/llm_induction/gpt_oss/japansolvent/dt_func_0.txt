def predict(row):
    # Precompute binary node values
    node1 = int(row['EBIT/TA'] <= 0.05)          # Root node
    node2 = int(row['Equity/TL'] <= 0.5)         # Child node
    emb = [node1, node2]

    # Decision tree logic (depth 2)
    if node1 == 1:  # EBIT/TA <= 0.05
        if node2 == 1:  # Equity/TL <= 0.5
            pred = 0  # Insolvent
        else:          # Equity/TL > 0.5
            pred = 1  # Solvent
    else:           # EBIT/TA > 0.05
        if node2 == 1:  # Equity/TL <= 0.5
            pred = 0  # Insolvent
        else:          # Equity/TL > 0.5
            pred = 1  # Solvent

    return pred, emb
