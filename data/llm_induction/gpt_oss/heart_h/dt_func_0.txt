def heart_disease_predict(row):
    # Precompute binary node evaluations
    node1 = int(row['age'] <= 50)          # Root node: age <= 50?
    node2 = int(row['trestbps'] <= 120)    # Left child split: resting BP <= 120?
    node3 = int(row['chol'] <= 200)        # Right child split: cholesterol <= 200?
    
    emb = [node1, node2, node3]
    
    # Decision tree logic
    if node1 == 1:          # age <= 50
        if node2 == 1:      # trestbps <= 120
            pred = 0       # No heart disease
        else:              # trestbps > 120
            pred = 1       # Heart disease
    else:                  # age > 50
        if node3 == 1:      # chol <= 200
            pred = 0       # No heart disease
        else:              # chol > 200
            pred = 1       # Heart disease
    
    return pred, emb
