def predict_penguin(row):
    # Precompute binary node evaluations
    node1 = int(row['flipper_length_mm'] <= 210)          # Root split
    node2 = int(row['body_mass_g'] <= 3500)               # Second level split (only used if node1 == 1)
    emb = [node1, node2]

    # Decision tree logic
    if node1 == 1:
        if node2 == 1:
            prediction = 0   # Adelie
        else:
            prediction = 1   # Chinstrap
    else:
        prediction = 2       # Gentoo

    return prediction, emb
