def credit_decision(row):
    # Precompute binary node evaluations
    node1 = int(row['Derogatory.reports'] <= 1)          # Root node
    node2 = int(row['Age'] <= 40)                        # Second level node
    emb = [node1, node2]

    # Decision tree logic
    if node1 == 1:
        prediction = 1
    else:
        if node2 == 1:
            prediction = 1
        else:
            prediction = 0

    return prediction, emb
