def predict_party(row):
    # Precompute binary embeddings for the inner nodes
    node1 = int(row['immigration'] == 2)          # 1 if voted yes, 0 if voted no
    emb = [node1]

    # Decision tree logic (depth 2: root node only)
    if node1 == 1:
