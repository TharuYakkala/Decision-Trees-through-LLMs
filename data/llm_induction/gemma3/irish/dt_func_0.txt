def predict_school_leaving_certificate(row):
    """
    Predicts whether a school leaving certificate is taken based on the provided features.

    Args:
        row (dict): A dictionary containing the feature values for a single observation.

    Returns:
        tuple: A tuple containing the prediction (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['DVRT'] > 60)
    node2 = int(row['Prestige_score'] > 15)
    node3 = int(row['Type_school'] == 1)
    node4 = int(row['Sex'] == 1)

    emb = [node1, node2, node3, node4]

    if node1 == 0:
        if node2 == 0:
            return 0, emb
        else:
            if node3 == 1:
                return 1, emb
            else:
                return 0, emb
    else:
        if node3 == 1:
            return 1, emb
        else:
            if node4 == 1:
                return 1, emb
            else:
                return 0, emb
