def classify_party(row):
    """
    Classifies party affiliation (Republican or Democrat) based on voting records.

    Args:
        row (dict): A dictionary containing the voting records for a single representative.

    Returns:
        tuple: A tuple containing the predicted party affiliation (0 for Democrat, 1 for Republican)
               and a list of binary embeddings representing the decision path.
    """
    node1 = int(row['aid-to-nicaraguan-contras'] > 1)
    node2 = int(row['immigration'] > 1)
    node3 = int(row['handicapped-infants'] > 1)
    emb = [node1, node2, node3]

    if node1 == 0:
        if node2 == 1:
            return 1, emb + [1]
        else:
            return 0, emb + [0]
    else:
        if node3 == 1:
            return 1, emb + [1]
        else:
            return 0, emb + [0]
