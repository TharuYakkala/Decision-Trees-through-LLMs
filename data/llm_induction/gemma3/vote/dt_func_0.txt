def classify_party_affiliation(row):
    """
    Classifies party affiliation (Republican or Democrat) based on voting records.

    Args:
        row (dict): A dictionary containing the voting records for a single legislator.

    Returns:
        tuple: A tuple containing the predicted party affiliation (0 for Democrat, 1 for Republican)
               and a list of binary node embeddings.
    """
    node1 = int(row['handicapped infants'] > 1)
    node2 = int(row['water project cost sharing'] > 1)
    node3 = int(row['aid to El Salvador'] > 1)
    node4 = int(row['immigration'] > 1)
    node5 = int(row['crime'] > 1)

    emb = [node1, node2, node3, node4, node5]

    if node1 == 0:
        if node2 == 1:
            return 1, emb + [1]
        else:
            if node3 == 1:
                return 1, emb + [2]
            else:
                if node4 == 0:
                    return 0, emb + [3]
                else:
                    if node5 == 1:
                        return 1, emb + [4]
                    else:
                        return 0, emb + [5]
    else:
        if node3 == 1:
            return 1, emb + [6]
        else:
            return 0, emb + [7]
