def predict_survival(row):
    """
    Predicts patient survival (0 = die, 1 = live) based on a decision tree.

    Args:
        row (dict): A dictionary containing the feature values for a patient.

    Returns:
        tuple: A tuple containing the predicted survival (0 or 1) and a list of binary node embeddings.
    """

    # Precompute binary node embeddings
    node1 = int(row['AGE'] > 50)
    node2 = int(row['SEX'] == 2)  # Male
    node3 = int(row['BILIRUBIN'] > 2.0)
    node4 = int(row['ALK PHOSPHATE'] > 200)
    node5 = int(row['PROTIME'] > 40)

    emb = [node1, node2, node3, node4, node5]

    # Decision tree logic
    if node1 == 0:
        if node2 == 1:  # Male
            if node3 == 1:  # Bilirubin > 2.0
                return 0, emb
            else:
                return 1, emb
        else:  # Female
            if node4 == 1:  # Alkaline Phosphatase > 200
                return 0, emb
            else:
                return 1, emb
    else:  # AGE > 50
        if node5 == 1:  # Prothrombin Time > 40
            return 0, emb
        else:
            return 1, emb
