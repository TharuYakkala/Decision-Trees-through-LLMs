def penguin_species_classifier(row):
    """
    Classifies penguin species based on a decision tree.

    Args:
        row (dict): A dictionary containing penguin features.

    Returns:
        tuple: A tuple containing the predicted species (0, 1, or 2) and a list of binary node embeddings.
    """

    # Precompute binary node embeddings
    node1 = int(row['island'] == 0)  # Biscoe
    node2 = int(row['culmen_length_mm'] > 35)
    node3 = int(row['flipper_length_mm'] > 200)
    node4 = int(row['body_mass_g'] > 4000)
    node5 = int(row['sex'] == 2) # male

    emb = [node1, node2, node3, node4, node5]

    # Decision tree logic
    if node1 == 0:  # Biscoe
        if node2 == 0:  # culmen_length_mm <= 35
            return 0, emb  # Adelie
        else:  # culmen_length_mm > 35
            if node3 == 0: # flipper_length_mm <= 200
                return 1, emb # Chinstrap
            else: # flipper_length_mm > 200
                return 2, emb # Gentoo
    else:  # Dream or Torgersen
        if node4 == 0: # body_mass_g <= 4000
            return 1, emb # Chinstrap
        else: # body_mass_g > 4000
            if node5 == 0: # sex != male
                return 2, emb # Gentoo
            else: # sex == male
                return 1, emb # Chinstrap
