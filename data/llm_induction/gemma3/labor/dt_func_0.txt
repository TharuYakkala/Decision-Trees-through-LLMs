def negotiate_outcome(row):
    """
    Predicts the outcome of labor negotiations based on provided features.

    Args:
        row (dict): A dictionary containing the feature values.

    Returns:
        tuple: A tuple containing the predicted class (0 or 1) and a list of binary node embeddings.
    """

    # Precompute binary vector embeddings
    node1 = int(row['duration'] <= 4)
    node2 = int(row['wage increase first year'] <= 3)
    node3 = int(row['cost of living adjustment'] != 'none')
    node4 = int(row['working hours'] > 38)
    node5 = int(row['pension'] != 'none')
    node6 = int(row['shift differential'] > 10)
    node7 = int(row['education allowance'] == 'true')
    node8 = int(row['statutory holidays'] >= 12)
    node9 = int(row['vacation'] == 'generous')
    node10 = int(row['longterm disability assistance'] == 'true')
    node11 = int(row['bereavement assistance'] == 'true')
    node12 = int(row['contribution to health plan'] != 'none')

    emb = [node1, node2, node3, node4, node5, node6, node7, node8, node9, node10, node11, node12]

    # Decision Tree Logic
    if node1 == 0:
        if node2 == 1:
            return 1, emb
        else:
            if node3 == 1:
                return 1, emb
            else:
                return 0, emb
    else:
        if node4 == 1:
            if node5 == 1:
                return 1, emb
            else:
                if node6 == 1:
                    return 1, emb
                else:
                    return 0, emb
        else:
            if node7 == 1:
                if node8 == 1:
                    return 1, emb
                else:
                    if node9 == 1:
                        return 1, emb
                    else:
                        return 0, emb
            else:
                if node10 == 1:
                    return 1, emb
                else:

                            return 1, emb
