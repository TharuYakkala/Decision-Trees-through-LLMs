def credit_application_decision_tree(row):
    """
    Predicts whether a credit application is accepted based on a decision tree.

    Args:
        row (dict): A dictionary containing the feature values for a single application.

    Returns:
        tuple: A tuple containing the prediction (0 or 1) and a list of binary node embeddings.
    """

    node1 = int(row['Income.per.dependent'] > 4)
    node2 = int(row['Own.home'] == 1)
    node3 = int(row['Derogatory.reports'] > 1)
    emb = [node1, node2, node3]

    if node1 == 1:
        if node2 == 1:
            return 1, emb
        else:
            if node3 == 1:
                return 0, emb
            else:
                return 1, emb
    else:
        if node2 == 1:
            if row['Monthly.credit.card.exp'] > 1000:
                return 1, emb
            else:
                return 0, emb
        else:
            if node3 == 1:
                return 0, emb
            else:
                return 0, emb
