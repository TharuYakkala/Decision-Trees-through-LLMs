def decision_tree(feature_row):
    node1 = int(feature_row['DVRT'] > 50)
    node2 = int(feature_row['Prestige_score'] > 15)
    node3 = int(feature_row['Educational_level'] <= 5)
    node4 = int(feature_row['Type_school'] == 1)

    emb = [node1, node2, node3, node4]

    if node1 and node2:
        prediction = 1
    elif node1 and not node2:
        prediction = 0
    elif not node1 and node3:
        prediction = 1
    elif not node1 and not node3 and node4:
        prediction = 0
    else:
        prediction = 1

    return prediction, emb
