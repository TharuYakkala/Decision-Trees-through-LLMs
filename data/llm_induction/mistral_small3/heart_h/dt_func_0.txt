def decision_tree(feature_row):
    node1 = int(feature_row['age'] > 50)
    node2 = int(feature_row['chest_pain'] == 1)
    node3 = int(feature_row['thalach'] < 150)
    node4 = int(feature_row['oldpeak'] > 2)
    node5 = int(feature_row['slope'] == 1)
    node6 = int(feature_row['ca'] > 1)
    node7 = int(feature_row['thal'] == 1)

    emb = [node1, node2, node3, node4, node5, node6, node7]

    if node1 and node2:
        prediction = 1
    elif node1 and not node2 and node3:
        prediction = 0
    elif node1 and not node2 and not node3 and node4:
        prediction = 1
    elif node1 and not node2 and not node3 and not node4 and node5:
        prediction = 0
    elif node1 and not node2 and not node3 and not node4 and not node5 and node6:
        prediction = 1
    elif node1 and not node2 and not node3 and not node4 and not node5 and not node6 and node7:
        prediction = 0
    else:
        prediction = 0

    return prediction, emb
