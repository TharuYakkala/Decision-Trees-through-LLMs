def predict_penguin_species(feature_row):
    node1 = int(feature_row['island'] == 2)
    node2 = int(feature_row['island'] == 1)
    node3 = int(feature_row['culmen_length_mm'] > 50)
    node4 = int(feature_row['flipper_length_mm'] > 200)
    node5 = int(feature_row['body_mass_g'] > 5000)
    node6 = int(feature_row['sex'] == 2)

    emb = [node1, node2, node3, node4, node5, node6]

    if node1:
        prediction = 2  # Gentoo
    elif node2:
        if node3 and node4:
            prediction = 1  # Chinstrap
        else:
            prediction = 0  # Adelie
    else:
        if node5 and node6:
            prediction = 2  # Gentoo
        else:
            prediction = 0  # Adelie

    return prediction, emb
